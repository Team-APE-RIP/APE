on_actions = {
    on_startup = {
		effect = {
			every_state = {
			    add_dynamic_modifier = { modifier = BuildingOutput_dynamic_modifiers }
			}
			every_possible_country = {
			    add_dynamic_modifier = { modifier = TagBuildingOutput_dynamic_modifiers }
			    add_offsite_building = {
				    type = industrial_complex
				    level = modifier@TagBuildingOutput
			    }
				set_variable = {
			        industrialComplexOffsiteBuildngsB = modifier@TagBuildingOutput
			    }
			}
		}
	}
	on_daily = {
	    effect = {
			add_offsite_building = {
				type = industrial_complex
				level = modifier@TagBuildingOutput
			}
	        multiply_variable = {
		        industrialComplexOffsiteBuildngsB = -1
		    }
			add_offsite_building = {
				type = industrial_complex
				level = industrialComplexOffsiteBuildngsB
			}
			set_variable = {
			    industrialComplexOffsiteBuildngsB = modifier@TagBuildingOutput
			}
		}
	}
}
