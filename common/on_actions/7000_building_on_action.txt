on_actions = {
    on_startup = {
		effect = {
			every_state = {
			    add_dynamic_modifier = { modifier = BuildingOutput_dynamic_modifiers }
			}
			every_possible_country = {
			    add_dynamic_modifier = { modifier = TagBuildingOutput_dynamic_modifiers }
			}
			every_possible_country = {
				set_variable = {
				    THIS.industrialComplexOffsiteBuildngsModifierValueRound = THIS.modifier@TagBuildingOutput
				}
				round_variable = THIS.industrialComplexOffsiteBuildngsModifierValueRound
			    add_offsite_building = {
				    type = industrial_complex
				    level = THIS.industrialComplexOffsiteBuildngsModifierValueRound
			    }
				add_to_array = {
				    THIS.industrialComplexOffsiteBuildngs = THIS.industrialComplexOffsiteBuildngsModifierValueRound
				}
				clear_variable = THIS.industrialComplexOffsiteBuildngsModifierValueRound
			}
		}
	}
	on_daily_ZZZ = {
	    effect = {
			every_possible_country = {
			    set_variable = {
				    THIS.industrialComplexOffsiteBuildngsModifierValueRound = THIS.modifier@TagBuildingOutput
				}
				round_variable = THIS.industrialComplexOffsiteBuildngsModifierValueRound
				add_to_array = {
				    THIS.industrialComplexOffsiteBuildngs = THIS.industrialComplexOffsiteBuildngsModifierValueRound
				}
				clear_variable = THIS.industrialComplexOffsiteBuildngsModifierValueRound
				set_variable = {
				    THIS.industrialComplexOffsiteBuildngsNum = THIS.industrialComplexOffsiteBuildngs^num
				}
				set_variable = {
				    THIS.industrialComplexOffsiteBuildngsA = THIS.industrialComplexOffsiteBuildngsNum
				}
				set_variable = {
				    THIS.industrialComplexOffsiteBuildngsB = THIS.industrialComplexOffsiteBuildngsNum
				}
				clear_variable = THIS.industrialComplexOffsiteBuildngsNum
				subtract_from_variable = {
				    THIS.industrialComplexOffsiteBuildngsA = 2
				}
				subtract_from_variable = {
				    THIS.industrialComplexOffsiteBuildngsB = 1
				}
				set_variable = {
				    THIS.industrialComplexOffsiteBuildngsValueA = THIS.industrialComplexOffsiteBuildngs^@var:THIS.industrialComplexOffsiteBuildngsA
				}
				set_variable = {
				    THIS.industrialComplexOffsiteBuildngsValueB = THIS.industrialComplexOffsiteBuildngs^@var:THIS.industrialComplexOffsiteBuildngsB
				}
				clear_variable = THIS.industrialComplexOffsiteBuildngsA
				clear_variable = THIS.industrialComplexOffsiteBuildngsB
				set_variable = {
				    THIS.industrialComplexOffsiteBuildngsSubtractValue = THIS.industrialComplexOffsiteBuildngsValueB
				}
				subtract_from_variable = {
				    THIS.industrialComplexOffsiteBuildngsSubtractValue = THIS.industrialComplexOffsiteBuildngsValueA
				}
				clear_variable = THIS.industrialComplexOffsiteBuildngsValueA
				clear_variable = THIS.industrialComplexOffsiteBuildngsValueB
				add_offsite_building = {
				    type = industrial_complex
				    level = THIS.industrialComplexOffsiteBuildngsSubtractValue
			    }
				clear_variable = THIS.industrialComplexOffsiteBuildngsSubtractValue
			}
		}
	}
}
