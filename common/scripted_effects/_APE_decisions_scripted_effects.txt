APE_refresh_decisions = {
    set_variable = { ROOT.APE_category_num = -1 }
	set_variable = { ROOT.APE_decision_num = 0 }
    for_each_loop = {
	    array = ROOT.APE_category
		value = value
		index = index
		set_variable = {
		    category_index = index
		}
		if = {
			limit = {
			    check_temp_variable = {
				    value = 1
				}
			}
			add_to_variable = {
			    ROOT.APE_category_num = 1
			}
		}
		set_variable = {
		    ROOT.APE_category_y^category_index = ROOT.APE_category_num
		}
		multiply_variable = {
		    ROOT.APE_category_y^category_index = 58
		}
		set_variable = {
		    ROOT.APE_decision_height = 40
		}
		multiply_variable = {
		    ROOT.APE_decision_height = ROOT.APE_decision_num
		}
		add_to_variable = {
		    ROOT.APE_category_y^category_index = ROOT.APE_decision_height
		}
		set_variable = { ROOT.APE_category_y_1^category_index = ROOT.APE_category_y^category_index }
		set_variable = { ROOT.APE_category_y_2^category_index = ROOT.APE_category_y^category_index }
		set_variable = { ROOT.APE_category_y_3^category_index = ROOT.APE_category_y^category_index }
		set_variable = { ROOT.APE_category_y_4^category_index = ROOT.APE_category_y^category_index }
		set_variable = { ROOT.APE_category_y_5^category_index = ROOT.APE_category_y^category_index }
		add_to_variable = {
		    ROOT.APE_category_y_2^category_index = 8
		}
		add_to_variable = {
		    ROOT.APE_category_y_3^category_index = 16
		}
		add_to_variable = {
		    ROOT.APE_category_y_4^category_index = 26
		}
		add_to_variable = {
		    ROOT.APE_category_y_5^category_index = 25
		}
		if = {
		    limit = {
			    check_variable = {
				    APE_decision_show^category_index = 1
				}
			}
			add_to_variable = {
		        ROOT.APE_decision_num = ROOT.APE_decision_category_@var:category_index^num
		    }
		}
		set_variable = {
		    ROOT.decision_bottom = ROOT.APE_category_y_4^category_index
		}
		set_variable = {
		    ROOT.APE_decision_height = 40
		}
		multiply_variable = {
		    ROOT.APE_decision_height = ROOT.APE_decision_category_@var:category_index^num
		}
		if = {
		    limit = {
			    check_variable = {
				    APE_decision_show^category_index = 1
				}
			}
		    add_to_variable = {
		        ROOT.decision_bottom = ROOT.APE_decision_height
		    }
		}
		add_to_variable = {
		    ROOT.decision_bottom = -8
		}
	}
	clear_variable = ROOT.APE_decision_num
	clear_variable = ROOT.APE_category_num
}
